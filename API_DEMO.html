前端 API 调用示例

TypeScript API Client

import { api } from '@/api/client'

// 登录
const result = await api.login('username', 'password')
api.setToken(result.access_token)

// 获取帖子列表
const posts = await api.listPosts(20, 0)

// 创建帖子
const post = await api.createPost({
  body: 'Hello World!',
  title: 'New Post'
})

数据集操作

// 启动数据生成
const job = await api.startGeneration({
  prompt: 'Generate travel data',
  count: 100,
  format: 'json'
})

// 获取任务状态
const status = await api.getJob(job.id)

// 导出数据
const export = await api.exportDataset({
  job_id: job.id,
  format: 'csv'
})
